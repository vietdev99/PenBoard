**2. Handle drag-connect mouse events in skia-canvas.tsx:**

In the existing `onMouseMove` handler, add a check at the TOP (before normal move handling):
```typescript
if (dragConnectState) {
  const rect = canvasContainerRef.current?.getBoundingClientRect()
  if (rect) {
    setMousePos({ x: e.clientX - rect.left, y: e.clientY - rect.top })
    // Hit test to find what element is under cursor
    const { viewport } = useCanvasStore.getState()
    const sceneX = (e.clientX - rect.left - viewport.panX) / viewport.zoom
    const sceneY = (e.clientY - rect.top - viewport.panY) / viewport.zoom
    const hitId = engine.current?.hitTest(sceneX, sceneY) ?? null
    setDragConnectHoveredId(hitId)
  }
  return  // Don't process normal mouse move during drag-connect
}
```

In the existing `onMouseUp` handler, add a check at the TOP:
```typescript
if (dragConnectState) {
  if (dragConnectHoveredId) {
    // Strip virtual ID prefix to get original child ID
    // Virtual IDs look like "refId__childId" -- we need the original childId
    const originalId = dragConnectHoveredId.includes('__')
      ? dragConnectHoveredId.split('__').pop()!
      : dragConnectHoveredId

    // Determine appropriate default target property based on arg type
    const argType = dragConnectState.argType
    const defaultProperty = getDefaultBindingProperty(argType, originalId)

    useDocumentStore.getState().addArgumentBinding(
      dragConnectState.sourceNodeId,
      dragConnectState.argId,
      { targetNodeId: originalId, targetProperty: defaultProperty }
    )
  }
  useCanvasStore.getState().setDragConnectState(null)
  setDragConnectHoveredId(null)
  return
}
```

Add helper function for default binding property:
```typescript
function getDefaultBindingProperty(argType: string, targetNodeId: string): string {
  // Get the target node to check its type
  const node = useDocumentStore.getState().getNodeById(targetNodeId)
  switch (argType) {
    case 'text': return node?.type === 'text' ? 'content' : 'name'
    case 'number': return 'opacity'
    case 'boolean': return 'visible'
    case 'select': return 'variant'
    case 'color': return 'fill.0.color'
    default: return 'visible'
  }
}
```

**3. Add global mouseup listener for drag-connect cancellation:**

In the SkiaCanvas component, add a useEffect that listens for mouseup on document when dragConnectState is active:
```tsx
useEffect(() => {
  if (!dragConnectState) return
  const handleGlobalMouseUp = () => {
    useCanvasStore.getState().setDragConnectState(null)
    setDragConnectHoveredId(null)
  }
  // Fallback: cancel on any mouseup outside canvas
  document.addEventListener('mouseup', handleGlobalMouseUp)
  return () => document.removeEventListener('mouseup', handleGlobalMouseUp)
}, [dragConnectState])
```
