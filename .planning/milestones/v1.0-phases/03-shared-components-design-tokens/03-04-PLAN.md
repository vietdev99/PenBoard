Add a grouping utility inside the component (or as a useMemo):

```typescript
// Group variables by type category
const groupedVariables = useMemo(() => {
  const entries = Object.entries(variables ?? {})
    .filter(([name]) => !search || name.toLowerCase().includes(search.toLowerCase()))
    .sort(([a], [b]) => a.localeCompare(b))

  const groups: Record<string, [string, VariableDefinition][]> = {
    Colors: [],
    Spacing: [],
    Typography: [],
    Other: [],
  }

  for (const entry of entries) {
    const [name, def] = entry
    if (def.type === 'color') {
      groups.Colors.push(entry)
    } else if (def.type === 'number') {
      // Heuristic: spacing vars have 'spacing', 'gap', 'padding', 'margin' in name
      // Typography vars have 'font', 'size', 'line', 'letter', 'text' in name
      const lowerName = name.toLowerCase()
      if (lowerName.includes('font') || lowerName.includes('size') || lowerName.includes('line') || lowerName.includes('letter') || lowerName.includes('text') || lowerName.includes('typo')) {
        groups.Typography.push(entry)
      } else {
        groups.Spacing.push(entry)
      }
    } else if (def.type === 'string') {
      const lowerName = name.toLowerCase()
      if (lowerName.includes('font')) {
        groups.Typography.push(entry)
      } else {
        groups.Other.push(entry)
      }
    } else {
      groups.Other.push(entry)
    }
  }

  return groups
}, [variables, search])
```

Add resolution for the new fields. Find the section in `resolveNodeForCanvas` where properties are resolved and add:

```typescript
// Resolve numeric variable refs on size fields
if ('width' in resolved && isVariableRef(resolved.width)) {
  const val = resolveVariableRef(String(resolved.width), variables, defaultTheme)
  if (val !== null && typeof val === 'number') resolved.width = val
}
if ('height' in resolved && isVariableRef(resolved.height)) {
  const val = resolveVariableRef(String(resolved.height), variables, defaultTheme)
  if (val !== null && typeof val === 'number') resolved.height = val
}

// Resolve corner radius
if ('cornerRadius' in resolved && typeof resolved.cornerRadius === 'string' && isVariableRef(resolved.cornerRadius)) {
  const val = resolveVariableRef(resolved.cornerRadius, variables, defaultTheme)
  if (val !== null && typeof val === 'number') resolved.cornerRadius = val
}

// Resolve text fields
if (resolved.type === 'text') {
  const textNode = resolved as any
  if (isVariableRef(textNode.fontSize)) {
    const val = resolveVariableRef(String(textNode.fontSize), variables, defaultTheme)
    if (val !== null && typeof val === 'number') textNode.fontSize = val
  }
  if (isVariableRef(textNode.lineHeight)) {
    const val = resolveVariableRef(String(textNode.lineHeight), variables, defaultTheme)
    if (val !== null && typeof val === 'number') textNode.lineHeight = val
  }
  if (isVariableRef(textNode.letterSpacing)) {
    const val = resolveVariableRef(String(textNode.letterSpacing), variables, defaultTheme)
    if (val !== null && typeof val === 'number') textNode.letterSpacing = val
  }
  if (isVariableRef(textNode.fontFamily)) {
    const val = resolveVariableRef(String(textNode.fontFamily), variables, defaultTheme)
    if (val !== null && typeof val === 'string') textNode.fontFamily = val
  }
}
```

**IMPORTANT:** The `resolveNodeForCanvas` function creates a shallow copy of the node (`{ ...node }`) at the top. All mutations should be on this copy, not the original. Verify this is the case before adding the new resolution code.

Also important: `isVariableRef` checks if a value is a string starting with `$`. For `width` and `height`, the type is `SizingBehavior = number | 'fit_content' | 'fill_container' | string`, so a `$variable` ref is a valid string value. The resolution should only replace it if the resolved value is a number.
